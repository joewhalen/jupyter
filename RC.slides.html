x = np.linspace(-5,5,10000)
omega = np.linspace(1,1.5,9)
#plt.figure(figsize=(12,6))
f = lambda x: np.sinc(x)
for i in omega:
    plt.plot(x,f(i*x),label='$\omega=$%3.3F'%i)
plt.legend()
plt.title('sinc($\omega x$)')
plt.xlabel('x')
plt.ylim(-0.4,1.1)
plt.legend(fontsize=10)
